# Python program to calculate the first 10,000 terms of the series
# 1 - 1/3 + 1/5 - 1/7 + ... and multiply the result by 4

total = 0.0
for i in range(10000):
    term = (-1)**i / (2*i + 1)  # Calculate each term: 1/(2i + 1) with alternating signs
    total += term               # Accumulate the total

result = total * 4  # Multiply the total by 4
print(result)  # Output the final result
